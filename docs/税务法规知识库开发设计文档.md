### 2.2 表结构设计

#### 2.2.1 用户表 (users)

```sql
CREATE TABLE users (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username        VARCHAR(50) UNIQUE NOT NULL,
    password_hash   VARCHAR(255) NOT NULL,
    role            VARCHAR(20) NOT NULL DEFAULT 'user',
    is_active       BOOLEAN NOT NULL DEFAULT TRUE,
    created_at      TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- 索引
CREATE INDEX idx_users_username ON users(username);
CREATE INDEX idx_users_role ON users(role);
```

#### 2.2.2 实体类型表 (entity_types)

```sql
CREATE TABLE entity_types (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name                VARCHAR(100) NOT NULL,
    description         TEXT,
    required_attributes JSONB NOT NULL DEFAULT '[]',
    optional_attributes JSONB NOT NULL DEFAULT '[]',
    is_system           BOOLEAN NOT NULL DEFAULT FALSE,
    created_at          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    created_by          UUID REFERENCES users(id)
);

-- 索引
CREATE INDEX idx_entity_types_name ON entity_types(name);
```

#### 2.2.3 关系类型表 (relation_types)

```sql
CREATE TABLE relation_types (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name            VARCHAR(100) NOT NULL,
    source_type     VARCHAR(100) NOT NULL,
    target_type     VARCHAR(100) NOT NULL,
    description     TEXT,
    is_system       BOOLEAN NOT NULL DEFAULT FALSE,
    created_at      TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    created_by      UUID REFERENCES users(id),
    FOREIGN KEY (source_type) REFERENCES entity_types(name),
    FOREIGN KEY (target_type) REFERENCES entity_types(name)
);

-- 索引
CREATE INDEX idx_relation_types_name ON relation_types(name);
CREATE INDEX idx_relation_types_source ON relation_types(source_type);
CREATE INDEX idx_relation_types_target ON relation_types(target_type);
```

#### 2.2.4 文档表 (documents)

```sql
CREATE TABLE documents (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    filename        VARCHAR(255) NOT NULL,
    original_name   VARCHAR(255) NOT NULL,
    file_path       VARCHAR(500) NOT NULL,
    file_type       VARCHAR(20) NOT NULL,
    file_size       BIGINT NOT NULL,
    status          VARCHAR(20) NOT NULL DEFAULT 'pending',
    parsed_text     TEXT,
    chunk_count     INTEGER,
    entity_count    INTEGER,
    relation_count  INTEGER,
    uploaded_by     UUID REFERENCES users(id),
    created_at      TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP

-- 索引
CREATE INDEX idx_documents_status ON documents(status);
CREATE INDEX idx_documents_uploaded_by ON documents(uploaded_by);
CREATE INDEX idx_documents_created_at ON documents(created_at DESC);
```

### 4.2 核心服务设计

#### 4.2.1 LightRAG Service

```python
# services/lightrag_service.py

from lightrag import LightRAG
from typing import List, Dict, Any

class LightRAGService:
    def __init__(self, working_dir: str):
        self.rag = LightRAG(
            working_dir=working_dir,
            # 配置 LLM、Embedding 等
        )

    async def insert_document(
        self,
        text: str,
        entity_schema: Dict[str, Any],
        metadata: Dict[str, Any] = None
    ) -> Dict[str, Any]:
        """
        插入文档，按自定义 Schema 抽取实体
        """
        # 1. 组装自定义抽取 Prompt
        prompt = self._build_extraction_prompt(text, entity_schema)
        # 2. 调用 LLM 抽取
        extraction_result = await self._extract_entities(prompt)
        # 3. 存入 LightRAG
        self.rag.insert(extraction_result["chunks"])
        return extraction_result

    def query(
        self,
        question: str,
        top_k: int = 10
    ) -> Dict[str, Any]:
        """
        问答查询
        """
        return self.rag.query(question, top_k=top_k)

    def _build_extraction_prompt(
        self,
        text: str,
        entity_schema: Dict[str, Any]
    ) -> str:
        """
        根据实体 Schema 组装抽取 Prompt
        """
        # 构建实体类型说明
        entity_types_desc = "\n".join([
            f"- {et['name']}: {et['description']}"
            for et in entity_schema["entity_types"]
        ])
        # 构建关系类型说明
        relation_types_desc = "\n".join([
            f"- {rt['name']}: {rt['description']} ({rt['source']} -> {rt['target']})"
            for rt in entity_schema["relation_types"]
        ])

        prompt = f"""
请从以下文本中抽取税务实体和关系。

## 实体类型
{entity_types_desc}

## 关系类型
{relation_types_desc}

## 待处理文本
{text}

请以 JSON 格式返回抽取结果。
"""
        return prompt
```

# 基于角色的权限装饰器

def require_role(role: str):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            user = get_current_user()
            if user.role != role and user.role != "admin":
                raise HTTPException(403, "权限不足")
            return await func(*args, **kwargs)
        return wrapper
